142  }
143 
144  // Init Gate Input
145  p->gate_in.pin = p->seed.GetPin(GATE_IN_PIN);
146  p->gate_in.mode = DSY_GPIO_MODE_INPUT;
147  p->gate_in.pull = DSY_GPIO_NOPULL;
148  dsy_gpio_init(&p->gate_in);
149 
150  // Init Gate Output
151  p->gate_out.pin = p->seed.GetPin(GATE_OUT_PIN);
152  p->gate_out.mode = DSY_GPIO_MODE_OUTPUT_PP;
153  p->gate_out.pull = DSY_GPIO_NOPULL;
154  dsy_gpio_init(&p->gate_out);
155 
156  /* Init LED Driver
157  2x PCA9685 addresses 0x00, and 0x01
158  TODO: add multidriver support
159  */
160  uint8_t addr[2] = {0x00, 0x02};
161 
162  dsy_led_driver_init(&p->seed.LED_DRIVER_I2C, addr, 2);
163 
164  //Init Keyboard Switches
165  //TODO: add cd4021 with parallel data support
166 
167  p->keyboard_sr.pin_config[DSY_SR_4021_PIN_CS]
168  = p->seed.GetPin(KB_SW_SR_CS_PIN);
169  p->keyboard_sr.pin_config[DSY_SR_4021_PIN_CLK]
170  = p->seed.GetPin(KB_SW_SR_CLK_PIN);
171  p->keyboard_sr.pin_config[DSY_SR_4021_PIN_DATA]
172  = p->seed.GetPin(KB_SW_SR_D1_PIN);
173  p->keyboard_sr.pin_config[DSY_SR_4021_PIN_DATA2]
174  = p->seed.GetPin(KB_SW_SR_D2_PIN);
175  p->keyboard_sr.num_daisychained = 1;
176  p->keyboard_sr.num_parallel = 2;
177  dsy_sr_4021_init(&p->keyboard_sr);
178 
179  // Init ADC (currently in daisy_seed).
180  AdcChannelConfig cfg[5];
181 
182  // Set up mux pin
183  cfg[0].InitMux(p->seed.GetPin(MUX_ADC_PIN),
184  8,
185  p->seed.GetPin(MUX_SEL_0_PIN),
186  p->seed.GetPin(MUX_SEL_1_PIN),
187  p->seed.GetPin(MUX_SEL_2_PIN));
188 
189  // Set up CV inputs
190  cfg[1].InitSingle(p->seed.GetPin(CV1_ADC_PIN));
191  cfg[2].InitSingle(p->seed.GetPin(CV2_ADC_PIN));
192  cfg[3].InitSingle(p->seed.GetPin(CV3_ADC_PIN));
193  cfg[4].InitSingle(p->seed.GetPin(CV4_ADC_PIN));
194 
195  // Init all 5 channels
196  p->seed.adc.Init(cfg, 5);
197 
198  // Setup Knob/CV Analog Controls Mapped to ADCs
199  for(size_t i = 0; i < KNOB_LAST; i++)
200  {
201  p->knobs[i].Init(p->seed.adc.GetMuxPtr(0, i), 1000.0f);
202  }
203  p->cvs[CV_1].InitBipolarCv(p->seed.adc.GetPtr(1), 1000.0f);
204  p->cvs[CV_2].InitBipolarCv(p->seed.adc.GetPtr(2), 1000.0f);
205  p->cvs[CV_3].InitBipolarCv(p->seed.adc.GetPtr(3), 1000.0f);
206  p->cvs[CV_4].InitBipolarCv(p->seed.adc.GetPtr(4), 1000.0f);
207 
208  // Start timer
209  dsy_tim_start();
210 }
