131 FORCE_INLINE void daisy_field_init(daisy_field *p)
132 {
133  uint8_t sw_pins[SW_LAST] = {SW_1_PIN, SW_2_PIN, SW_3_PIN};
134  // dsy_gpio_port sw_ports[SW_LAST] = {SW_1_PORT, SW_2_PORT, SW_3_PORT};
135 
136  // Init Daisy Seed
137  p->seed.Configure();
138  p->seed.Init();
139 
140  // Init Switches
141  for(uint8_t i = 0; i < SW_LAST; i++)
142  {
143  p->switches[i].Init(p->seed.GetPin(sw_pins[i]), 1000.0f);
144  }
145 
146  // Init Gate Input
147  p->gate_in.pin = p->seed.GetPin(GATE_IN_PIN);
148  p->gate_in.mode = DSY_GPIO_MODE_INPUT;
149  p->gate_in.pull = DSY_GPIO_NOPULL;
150  dsy_gpio_init(&p->gate_in);
151 
152  // Init Gate Output
153  p->gate_out.pin = p->seed.GetPin(GATE_OUT_PIN);
154  p->gate_out.mode = DSY_GPIO_MODE_OUTPUT_PP;
155  p->gate_out.pull = DSY_GPIO_NOPULL;
156  dsy_gpio_init(&p->gate_out);
157 
158  /* Init LED Driver
159  2x PCA9685 addresses 0x00, and 0x01
160  TODO: add multidriver support
161  */
162  uint8_t addr[2] = {0x00, 0x02};
163 
164  dsy_led_driver_init(&p->seed.LED_DRIVER_I2C, addr, 2);
165 
166  /* Init Keyboard Switches
167  TODO: add cd4021 with parallel data support
168  */
169  p->keyboard_sr.pin_config[DSY_SR_4021_PIN_CS]
170  = p->seed.GetPin(KB_SW_SR_CS_PIN);
171  p->keyboard_sr.pin_config[DSY_SR_4021_PIN_CLK]
172  = p->seed.GetPin(KB_SW_SR_CLK_PIN);
173  p->keyboard_sr.pin_config[DSY_SR_4021_PIN_DATA]
174  = p->seed.GetPin(KB_SW_SR_D1_PIN);
175  p->keyboard_sr.pin_config[DSY_SR_4021_PIN_DATA2]
176  = p->seed.GetPin(KB_SW_SR_D2_PIN);
177  p->keyboard_sr.num_daisychained = 1;
178  p->keyboard_sr.num_parallel = 2;
179  dsy_sr_4021_init(&p->keyboard_sr);
180 
181  // Init ADC (currently in daisy_seed).
182  AdcChannelConfig cfg[5];
183 
184  // Set up mux pin
185  cfg[0].InitMux(p->seed.GetPin(MUX_ADC_PIN),
186  p->seed.GetPin(MUX_SEL_0_PIN),
187  p->seed.GetPin(MUX_SEL_1_PIN),
188  p->seed.GetPin(MUX_SEL_2_PIN),
189  8);
190  // Set up CV inputs
191  cfg[1].InitSingle(p->seed.GetPin(CV1_ADC_PIN));
192  cfg[2].InitSingle(p->seed.GetPin(CV2_ADC_PIN));
193  cfg[3].InitSingle(p->seed.GetPin(CV3_ADC_PIN));
194  cfg[4].InitSingle(p->seed.GetPin(CV4_ADC_PIN));
195 
196  // Init all 5 channels
197  p->seed.adc.Init(cfg, 5);
198 
199  // Setup Knob/CV Analog Controls Mapped to ADCs
200  for(size_t i = 0; i < KNOB_LAST; i++)
201  {
202  p->knobs[i].Init(p->seed.adc.GetMuxPtr(0, i), 1000.0f);
203  }
204  p->cvs[CV_1].InitBipolarCv(p->seed.adc.GetPtr(1), 1000.0f);
205  p->cvs[CV_2].InitBipolarCv(p->seed.adc.GetPtr(2), 1000.0f);
206  p->cvs[CV_3].InitBipolarCv(p->seed.adc.GetPtr(3), 1000.0f);
207  p->cvs[CV_4].InitBipolarCv(p->seed.adc.GetPtr(4), 1000.0f);
208 
209  // Start timer
210  dsy_tim_start();
211 }
